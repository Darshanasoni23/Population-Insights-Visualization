import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset again, but set the correct header
file_path = "project.csv"
df = pd.read_csv(file_path, skiprows=3, header=0)  # Set row 3 as header

# Rename the columns properly
df.rename(columns={df.columns[0]: 'Country Name', df.columns[-1]: '2023'}, inplace=True)

# Select relevant columns
df = df[['Country Name', '2023']]

# Drop missing values
df = df.dropna()

# Convert population column to numeric
df['2023'] = pd.to_numeric(df['2023'], errors='coerce')

# Sort and select top 10 populated countries
df = df.sort_values(by='2023', ascending=False).head(10)

# Bar Chart: Top 10 Populated Countries
plt.figure(figsize=(10, 6))
sns.barplot(x='2023', y='Country Name', data=df, palette='viridis')
plt.xlabel('Population (2023)')
plt.ylabel('Country')
plt.title('Top 10 Most Populated Countries in 2023')
plt.show()

# Histogram: Distribution of Population
plt.figure(figsize=(10, 6))
sns.histplot(df['2023'], bins=10, kde=True, color='blue')
plt.xlabel('Population')
plt.ylabel('Number of Countries')
plt.title('Population Distribution (2023)')
plt.show()
